112 void corenrn_parameters::parse (int argc, char** argv) {
113 
114  try {
115  app.parse(argc, argv);
116  if(verbose == verbose_level::NONE) {
117  nrn_nobanner_ = 1;
118  }
119  } catch (const CLI::ExtrasError &e) {
120  // in case of parsing errors, show message with exception
121  std::cerr << "CLI parsing error, see nrniv-core --help for more information. \n" << std::endl;
122  app.exit(e);
123  throw e;
124 
125  } catch (const CLI::ParseError &e) {
126  // use --help is also ParseError; in this case exit by showing all options
127  app.exit(e);
128  exit(0);
129  }
130 
131  // is user has asked for version info, print it and exit
132  if (show_version) {
133  std::cout << "CoreNEURON Version : " << cnrn_version() << std::endl;
134  exit(0);
135  }
136 };
