1 /*
2 Permute nodes.
3 
4 To make gaussian elimination on gpu more efficient.
5 
6 Permutation vector p[i] applied to a data vector, moves the data_original[i]
7 to data[p[i]].
8 That suffices for node properties such as area[i], a[i], b[i]. e.g.
9  area[p[i]] <- area_original[i]
10 
11 Notice that p on the left side is a forward permutation. On the right side
12 it serves as the inverse permutation.
13 area_original[i] <- area_permuted[p[i]]
14 
15 but things
16 get a bit more complicated when the data is an integer index into the
17 original data.
18 
19 For example:
20 
21 parent[i] needs to be transformed so that
22 parent[p[i]] <- p[parent_original[i]] except that if parent_original[j] = -1
23  then parent[p[j]] = -1
24 
25 membrane mechanism nodelist ( a subset of nodes) needs to be at least
26 minimally transformed so that
27 nodelist_new[k] <- p[nodelist_original[k]]
28 This does not affect the order of the membrane mechanism property data.
29 
30 However, computation is more efficient to permute (sort) nodelist_new so that
31 it follows as much as possible the permuted node ordering, ie in increasing
32 node order. Consider this further mechanism specific nodelist permutation,
33 which is to be applied to the above nodelist_new, to be p_m, which has the same
34 size as nodelist. ie.
35 nodelist[p_m[k]] <- nodelist_new[k].
36 
37 Notice the similarity to the parent case...
38 nodelist[p_m[k]] = p[nodelist_original[k]]
39 
40 and now the membrane mechanism node data, does need to be permuted to have an
41 order consistent with the new nodelist. Since there are nm instances of the
42 mechanism each with sz data values (consider AoS layout).
43 The data permutation is
44 for k=[0:nm] for isz=[0:sz]
45  data_m[p_m[k]*sz + isz] = data_m_original[k*sz + isz]
46 
47 For an SoA layout the indexing is k + isz*nm (where nm may include padding).
48 
49 A more complicated case is a mechanisms dparam array (nm instances each with
50 dsz values) Some of those values are indices into another mechanism (eg
51 pointers to ion properties) or voltage or area depending on the semantics of
52 the value. We can use the above data_m permutation but then need to update
53 the values according to the permutation of the object the value indexes into.
54 Consider the permutation of the target object to be p_t . Then a value
55 iold = pdata_m(k, isz) - data_t in AoS format
56 refers to k_t = iold % sz_t and isz_t = iold - k_t*sz_t
57 and for a target in SoA format isz_t = iold % nm_t and k_t = iold - isz_t*nm_t
58 ie k_t_new = p_m_t[k_t] so, for AoS, inew = k_t_new*sz_t + isz_t
59 or , for SoA, inew = k_t_new + isz_t*nm_t
60 so pdata_m(k, isz) = inew + data_t
61 
62 
63 */

76 void permute(T* data, int cnt, int sz, int layout, int* p) {
77  // data(p[icnt], isz) <- data(icnt, isz)
78  // this does not change data, merely permutes it.
79  // assert len(p) == cnt
80  if (!p) {
81  return;
82  }
83  int n = cnt * sz;
84  if (n < 1) {
85  return;
86  }
87 
88  if (layout == 0) { // for SoA, n might be larger due to cnt padding
89  n = nrn_soa_padded_size(cnt, layout) * sz;
90  }
91 
92  T* data_orig = new T[n];
93  for (int i = 0; i < n; ++i) {
94  data_orig[i] = data[i];
95  }
96 
97  for (int icnt = 0; icnt < cnt; ++icnt) {
98  for (int isz = 0; isz < sz; ++isz) {
99  // note that when layout==0, nrn_i_layout takes into account SoA padding.
100  int i = nrn_i_layout(icnt, cnt, isz, sz, layout);
101  int ip = nrn_i_layout(p[icnt], cnt, isz, sz, layout);
102  data[ip] = data_orig[i];
103  }
104  }
105 
106  delete[] data_orig;
107 }

125 void update_pdata_values(Memb_list* ml, int type, NrnThread& nt) {
126  // assumes AoS to SoA transformation already made since we are using
127  // nrn_i_layout to determine indices into both ml->pdata and into target data
128  int psz = corenrn.get_prop_dparam_size()[type];
129  if (psz == 0) {
130  return;
131  }
132  if (corenrn.get_is_artificial()[type]) {
133  return;
134  }
135  int* semantics = corenrn.get_memb_func(type).dparam_semantics;
136  if (!semantics) {
137  return;
138  }
139  int* pdata = ml->pdata;
140  int layout = corenrn.get_mech_data_layout()[type];
141  int cnt = ml->nodecount;
142  // ml padding does not matter (but target padding does matter)
143 
144  // interesting semantics are -1 (area), -5 (pointer), -9 (diam), or 0-999 (ion variables)
145  for (int i = 0; i < psz; ++i) {
146  int s = semantics[i];
147  if (s == -1) { // area
148  int area0 = nt._actual_area - nt._data; // includes padding if relevant
149  int* p_target = nt._permute;
150  for (int iml = 0; iml < cnt; ++iml) {
151  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
152  // *pd is the original integer into nt._data . Needs to be replaced
153  // by the permuted value
154 
155  // This is ok whether or not area changed by padding?
156  // since old *pd updated appropriately by earlier AoS to SoA
157  // transformation
158  int ix = *pd - area0; // original integer into area array.
159  nrn_assert((ix >= 0) && (ix < nt.end));
160  int ixnew = p_target[ix];
161  *pd = ixnew + area0;
162  }
163  } else if (s == -9) { // diam
164  int diam0 = nt._actual_diam - nt._data; // includes padding if relevant
165  int* p_target = nt._permute;
166  for (int iml = 0; iml < cnt; ++iml) {
167  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
168  // *pd is the original integer into nt._data . Needs to be replaced
169  // by the permuted value
170 
171  // This is ok whether or not diam changed by padding?
172  // since old *pd updated appropriately by earlier AoS to SoA
173  // transformation
174  int ix = *pd - diam0; // original integer into actual_diam array.
175  nrn_assert((ix >= 0) && (ix < nt.end));
176  int ixnew = p_target[ix];
177  *pd = ixnew + diam0;
178  }
179  } else if (s == -5) { // assume pointer to membrane voltage
180  int v0 = nt._actual_v - nt._data;
181  // same as for area semantics
182  int* p_target = nt._permute;
183  for (int iml = 0; iml < cnt; ++iml) {
184  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
185  int ix = *pd - v0; // original integer into area array.
186  nrn_assert((ix >= 0) && (ix < nt.end));
187  int ixnew = p_target[ix];
188  *pd = ixnew + v0;
189  }
190  } else if (s >= 0 && s < 1000) { // ion
191  int etype = s;
192  int elayout = corenrn.get_mech_data_layout()[etype];
193  Memb_list* eml = nt._ml_list[etype];
194  int edata0 = eml->data - nt._data;
195  int ecnt = eml->nodecount;
196  int esz = corenrn.get_prop_param_size()[etype];
197  int* p_target = eml->_permute;
198  for (int iml = 0; iml < cnt; ++iml) {
199  int* pd = pdata + nrn_i_layout(iml, cnt, i, psz, layout);
200  int ix = *pd - edata0;
201  // from ix determine i_ecnt and i_esz (need to permute i_ecnt)
202  int i_ecnt, i_esz, padded_ecnt;
203  if (elayout == 1) { // AoS
204  padded_ecnt = ecnt;
205  i_ecnt = ix / esz;
206  i_esz = ix % esz;
207  } else { // SoA
208  assert(elayout == 0);
209  padded_ecnt = nrn_soa_padded_size(ecnt, elayout);
210  i_ecnt = ix % padded_ecnt;
211  i_esz = ix / padded_ecnt;
212  }
213  int i_ecnt_new = p_target[i_ecnt];
214  int ix_new = nrn_i_layout(i_ecnt_new, ecnt, i_esz, esz, elayout);
215  *pd = ix_new + edata0;
216  }
217  }
218  }

247 int nrn_index_permute(int ix, int type, Memb_list* ml) {
248  int* p = ml->_permute;
249  if (!p) {
250  return ix;
251  }
252  int layout = corenrn.get_mech_data_layout()[type];
253  if (layout == 1) {
254  int sz = corenrn.get_prop_param_size()[type];
255  int i_cnt = ix / sz;
256  int i_sz = ix % sz;
257  return p[i_cnt] * sz + i_sz;
258  } else {
259  assert(layout == 0);
260  int padded_cnt = nrn_soa_padded_size(ml->nodecount, layout);
261  int i_cnt = ix % padded_cnt;
262  int i_sz = ix / padded_cnt;
263  return i_sz * padded_cnt + p[i_cnt];
264  }

301 int* nrn_index_sort(int* values, int n) {
302  std::vector<std::pair<int, int> > vi(n);
303  for (int i = 0; i < n; ++i) {
304  vi[i].first = values[i];
305  vi[i].second = i;
306  }
307  std::sort(vi.begin(), vi.end(), nrn_index_sort_cmp);
308  int* sort_indices = new int[n];
