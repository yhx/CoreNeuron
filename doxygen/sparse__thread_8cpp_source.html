148  int n,
149  int* s,
150  int* d,
151  double* t,
152  double dt,
153  SPFUN fun,
154  int linflag,
155  _threadargsproto_) {
156 #define ix(arg) ((arg)*_STRIDE)
157 #define s_(arg) _p[ix(s[arg])]
158 #define d_(arg) _p[ix(d[arg])]
159 
160  int i, j, ierr;
161  double err;
162 
163  for (i = 0; i < n; i++) { /*save old state*/
164  d_(i) = s_(i);
165  }
166  for (err = 1, j = 0; err > CONVERGE; j++) {
167  init_coef_list(so, _iml);
168  spfun(fun, so, so->rhs);
169  if ((ierr = matsol(so, _iml))) {
170  return ierr;
171  }
172  for (err = 0., i = 1; i <= n; i++) { /* why oh why did I write it from 1 */
173  s_(i - 1) += so->rhs[ix(i)];
174  if (!linflag && s_(i - 1) < 0.) {
175  s_(i - 1) = 0.;
176  }
177  err += fabs(so->rhs[ix(i)]);
178  }
179  if (j > MAXSTEPS) {
180  return EXCEED_ITERS;
181  }
182  if (linflag)
183  break;
184  }
185  init_coef_list(so, _iml);
186  spfun(fun, so, so->rhs);
187  for (i = 0; i < n; i++) { /*restore Dstate at t+dt*/
188  d_(i) = (s_(i) - d_(i)) / dt;
189  }
190  return SUCCESS;
191 }

218 static int matsol(SparseObj* so, int _iml) {
219  /* Upper triangularization */
220  so->numop = 0;
221  for (unsigned i = 1; i <= so->neqn; i++) {
222  Elm *pivot;
223  if (fabs((pivot = so->diag[i])->value[_iml]) <= ROUNDOFF) {
224  return SINGULAR;
225  }
226  /* Eliminate all elements in pivot column */
227  for (auto el = pivot->r_down; el; el = el->r_down) {
228  subrow(so, pivot, el, _iml);
229  }
230  }
231  bksub(so, _iml);
232  return (SUCCESS);
233 }

235 static void subrow(SparseObj* so, Elm* pivot, Elm* rowsub, int _iml) {
236  int _cntml_padded = so->_cntml_padded;
237  double r = rowsub->value[_iml] / pivot->value[_iml];
238  so->rhs[ix(rowsub->row)] -= so->rhs[ix(pivot->row)] * r;
239  so->numop++;
240  for (auto el = pivot->c_right; el; el = el->c_right) {
241  for (rowsub = rowsub->c_right; rowsub->col != el->col; rowsub = rowsub->c_right) {
242  ;
243  }
244  rowsub->value[_iml] -= el->value[_iml] * r;
245  so->numop++;
246  }

250  int _cntml_padded = so->_cntml_padded;

252  for (Elm* el = so->diag[i]->c_right; el; el = el->c_right) {

316 static Elm* getelm(SparseObj* so, unsigned row, unsigned col, Elm* new_elem)
317 /* return pointer to row col element maintaining order in rows */
318 {
319  Elm *el, *elnext;
320 
321  unsigned vrow = so->varord[row];
322  unsigned vcol = so->varord[col];
323 
324  if (vrow == vcol) {
325  return so->diag[vrow]; /* a common case */
326  }
327  if (vrow > vcol) { /* in the lower triangle */
328  /* search downward from diag[vcol] */
329  for (el = so->diag[vcol];; el = elnext) {
330  elnext = el->r_down;
331  if (!elnext) {
332  break;
333  } else if (elnext->row == row) { /* found it */
334  return elnext;
335  } else if (so->varord[elnext->row] > vrow) {
336  break;
337  }
338  }
339  /* insert below el */
340  if (!new_elem) {
341  new_elem = (Elm*)nrn_pool_alloc(so->elmpool);
342  new_elem->value = (double*)ecalloc(so->_cntml_padded, sizeof(double));
343  increase_order(so, row);
344  }
345  new_elem->r_down = el->r_down;
346  el->r_down = new_elem;
347  new_elem->r_up = el;
348  if (new_elem->r_down) {
349  new_elem->r_down->r_up = new_elem;
350  }
351  /* search leftward from diag[vrow] */
352  for (el = so->diag[vrow];; el = elnext) {
353  elnext = el->c_left;
354  if (!elnext) {
355  break;
356  } else if (so->varord[elnext->col] < vcol) {
357  break;
358  }
359  }
360  /* insert to left of el */
361  new_elem->c_left = el->c_left;
362  el->c_left = new_elem;
363  new_elem->c_right = el;
364  if (new_elem->c_left) {
365  new_elem->c_left->c_right = new_elem;
366  } else {
367  so->rowst[vrow] = new_elem;
368  }
369  } else { /* in the upper triangle */
370  /* search upward from diag[vcol] */
371  for (el = so->diag[vcol];; el = elnext) {
372  elnext = el->r_up;
373  if (!elnext) {
374  break;
375  } else if (elnext->row == row) { /* found it */
376  return elnext;
377  } else if (so->varord[elnext->row] < vrow) {
378  break;
379  }
380  }
381  /* insert above el */
382  if (!new_elem) {
383  new_elem = (Elm*)nrn_pool_alloc(so->elmpool);
384  new_elem->value = (double*)ecalloc(so->_cntml_padded, sizeof(double));
385  increase_order(so, row);
386  }
387  new_elem->r_up = el->r_up;
388  el->r_up = new_elem;
389  new_elem->r_down = el;
390  if (new_elem->r_up) {
391  new_elem->r_up->r_down = new_elem;
392  }
393  /* search right from diag[vrow] */
394  for (el = so->diag[vrow];; el = elnext) {
395  elnext = el->c_right;
396  if (!elnext) {
397  break;
398  } else if (so->varord[elnext->col] > vcol) {
399  break;
400  }
401  }
402  /* insert to right of el */
403  new_elem->c_right = el->c_right;
404  el->c_right = new_elem;
405  new_elem->c_left = el;
406  if (new_elem->c_right) {
407  new_elem->c_right->c_left = new_elem;
408  }
409  }
410  new_elem->row = row;
411  new_elem->col = col;
412  return new_elem;
413 }

521 static void get_next_pivot(SparseObj* so, unsigned i) {
522  /* get varord[i], etc. from the head of the orderlist. */
523  Item* order = so->orderlist->next;
524  assert(order != so->orderlist);
525 
526  unsigned j;
527  if ((j = so->varord[order->elm->row]) != i) {
528  /* push order lists down by 1 and put new diag in empty slot */
529  assert(j > i);
530  Elm* el = so->rowst[j];
531  for (; j > i; j--) {
532  so->diag[j] = so->diag[j - 1];
533  so->rowst[j] = so->rowst[j - 1];
534  so->varord[so->diag[j]->row] = j;
535  }
536  so->diag[i] = order->elm;
537  so->rowst[i] = el;
538  so->varord[so->diag[i]->row] = i;
539  /* at this point row links are out of order for diag[i]->col
540  and col links are out of order for diag[i]->row */
541  re_link(so, i);
542  }
543 
544  /* now make sure all needed elements exist */
545  for (Elm* el = so->diag[i]->r_down; el; el = el->r_down) {
546  for (Elm* pivot = so->diag[i]->c_right; pivot; pivot = pivot->c_right) {
547  IGNORE(getelm(so, el->row, pivot->col, ELM0));
548  }
549  reduce_order(so, el->row);
550  }
551 
552 #if DEBUG
553 {int j; Item *_or;
554  printf("%d ", i);
555  for (_or = so->orderlist->next, j=0; j<5 && _or != so->orderlist; j++, _or=_or->next) {
556  printf("(%d, %d) ", _or->elm->row, _or->norder);
557  }
558  printf("\n");
559 }
560 #endif
561  delete_item(order);
562 }

648 static void check_assert(SparseObj* so) {
649  /* check that all links are consistent */
650  for (unsigned i = 1; i <= so->neqn; i++) {
651  assert(so->diag[i]);

653  assert(so->varord[so->diag[i]->row] == i);

674 /* at this point row links are out of order for diag[i]->col
675  and col links are out of order for diag[i]->row */
676 static void re_link(SparseObj* so, unsigned i) {
677  for (Elm* el = so->rowst[i]; el; el = el->c_right) {
678  /* repair hole */
679  if (el->r_up)
680  el->r_up->r_down = el->r_down;
681  if (el->r_down)
682  el->r_down->r_up = el->r_up;
683  }
684 
685  for (Elm* el = so->diag[i]->r_down; el; el = el->r_down) {
686  /* repair hole */
687  if (el->c_right)
688  el->c_right->c_left = el->c_left;
689  if (el->c_left)
690  el->c_left->c_right = el->c_right;
691  else
692  so->rowst[so->varord[el->row]] = el->c_right;
693  }
694 
695  for (Elm* el = so->diag[i]->r_up; el; el = el->r_up) {
696  /* repair hole */
697  if (el->c_right)
698  el->c_right->c_left = el->c_left;
699  if (el->c_left)
700  el->c_left->c_right = el->c_right;
701  else
702  so->rowst[so->varord[el->row]] = el->c_right;
703  }
704 
705  /* matrix is consistent except that diagonal row elements are unlinked from
706  their columns and the diagonal column elements are unlinked from their
707  rows.
708  For simplicity discard all knowledge of links and use getelm to relink
709  */
710  Elm *dright, *dleft, *dup, *ddown, *elnext;
711 
712  so->rowst[i] = so->diag[i];
713  dright = so->diag[i]->c_right;
714  dleft = so->diag[i]->c_left;
715  dup = so->diag[i]->r_up;
716  ddown = so->diag[i]->r_down;
717  so->diag[i]->c_right = so->diag[i]->c_left = ELM0;
718  so->diag[i]->r_up = so->diag[i]->r_down = ELM0;
719  for (Elm* el = dright; el; el = elnext) {
720  elnext = el->c_right;
721  IGNORE(getelm(so, el->row, el->col, el));
722  }
723  for (Elm* el = dleft; el; el = elnext) {
724  elnext = el->c_left;
725  IGNORE(getelm(so, el->row, el->col, el));
726  }
727  for (Elm* el = dup; el; el = elnext) {
728  elnext = el->r_up;
729  IGNORE(getelm(so, el->row, el->col, el));
730  }
731  for (Elm* el = ddown; el; el = elnext) {
732  elnext = el->r_down;
733  IGNORE(getelm(so, el->row, el->col, el));
734  }
735 }
