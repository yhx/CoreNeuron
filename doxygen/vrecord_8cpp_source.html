142 double VecPlayContinuous::interpolate(double tt) {
143  if (tt >= t_[ubound_index_]) {
144  last_index_ = ubound_index_;
145  if (last_index_ == 0) {
146  // printf("return last tt=%g ubound=%g y=%g\n", tt, t_->elem(ubound_index_),
147  // y_->elem(last_index_));
148  return y_[last_index_];
149  }
150  } else if (tt <= t_[0]) {
151  last_index_ = 0;
152  // printf("return elem(0) tt=%g t0=%g y=%g\n", tt, t_->elem(0), y_->elem(0));
153  return y_[0];
154  } else {
155  search(tt);
156  }
157  double x0 = y_[last_index_ - 1];
158  double x1 = y_[last_index_];
159  double t0 = t_[last_index_ - 1];
160  double t1 = t_[last_index_];
161  // printf("IvocVectRecorder::continuous tt=%g t0=%g t1=%g theta=%g x0=%g x1=%g\n", tt, t0, t1,
162  // (tt - t0)/(t1 - t0), x0, x1);
163  if (t0 == t1) {
164  return (x0 + x1) / 2.;
165  }
166  return interp((tt - t0) / (t1 - t0), x0, x1);
167 }
