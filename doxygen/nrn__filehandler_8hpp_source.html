128  /** Parse a neuron section segment mapping
129  *
130  * Read count no of mappings for section to segment
131  */
132  template <typename T>
133  int read_mapping_info(T* mapinfo) {
134  int nsec, nseg, n_scan;
135  char line_buf[max_line_length], name[max_line_length];
136 
137  F.getline(line_buf, sizeof(line_buf));
138  n_scan = sscanf(line_buf, "%s %d %d", name, &nsec, &nseg);
139 
140  nrn_assert(n_scan == 3);
141 
142  mapinfo->name = std::string(name);
143 
144  if (nseg) {
145  std::vector<int> sec, seg;
146  sec.reserve(nseg);
147  seg.reserve(nseg);
148 
149  read_array<int>(&sec[0], nseg);
150  read_array<int>(&seg[0], nseg);
151 
152  for (int i = 0; i < nseg; i++) {
153  mapinfo->add_segment(sec[i], seg[i]);
154  }
155  }
156  return nseg;
157  }

162  /** Generic parse function for an array of fixed length.
163  *
164  * \tparam T the array element type: may be \c int or \c double.
165  * \param p pointer to the target in memory for reading into.
166  * \param count number of items of type \a T to parse.
167  * \param action whether to validate and skip (\c seek) or
168  * copy array into memory (\c read).
169  * \return the supplied pointer value.
170  *
171  * Error if \a count is non-zero, \a flag is \c read, and
172  * the supplied pointer \p is null.
173  *
174  * Arrays are represented by a checkpoint line followed by
175  * the array items in increasing index order, in the native binary
176  * representation of the writing process.
177  */
178  template <typename T>
179  inline T* parse_array(T* p, size_t count, parse_action flag) {
180  if (count > 0 && flag != seek)
181  nrn_assert(p != 0);
182 
183  read_checkpoint_assert();
184  switch (flag) {
185  case seek:
186  F.seekg(count * sizeof(T), std::ios_base::cur);
187  break;
188  case read:
189  F.read((char*)p, count * sizeof(T));
190  break;
191  }
192 
193  nrn_assert(!F.fail());
194  return p;
195  }

231  /** Write a padded array. nb_elements is number of elements to write per line,
232  * line_width is full size of a line in nb elements**/
233  template <typename T>
234  void write_array(T* p,
235  size_t nb_elements,
236  size_t line_width,
237  size_t nb_lines,
238  bool to_transpose = false) {
239  nrn_assert(F.is_open());
240  nrn_assert(current_mode & std::ios::out);
241  write_checkpoint();
242  T* temp_cpy = new T[nb_elements * nb_lines];
243 
244  if (to_transpose) {
245  for (size_t i = 0; i < nb_lines; i++) {
246  for (size_t j = 0; j < nb_elements; j++) {
247  temp_cpy[i + j * nb_lines] = p[i * line_width + j];
248  }
249  }
250  } else {
251  memcpy(temp_cpy, p, nb_elements * sizeof(T) * nb_lines);
252  }
253  // AoS never use padding, SoA is translated above, so one write
254  // operation is enought in both cases
255  F.write((const char*)temp_cpy, nb_elements * sizeof(T) * nb_lines);
256  nrn_assert(!F.fail());
257  delete[] temp_cpy;
258  }
