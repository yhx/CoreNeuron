87 void local_spikevec_sort(std::vector<double>& isvect,
88  std::vector<int>& isvecg,
89  std::vector<double>& osvect,
90  std::vector<int>& osvecg) {
91  osvect.resize(isvect.size());
92  osvecg.resize(isvecg.size());
93  // first build a permutation vector
94  std::vector<std::size_t> perm(isvect.size());
95  std::iota(perm.begin(), perm.end(), 0);
96  // sort by gid (second predicate first)
97  std::stable_sort(perm.begin(), perm.end(),
98  [&](std::size_t i, std::size_t j) { return isvecg[i] < isvecg[j]; });
99  // then sort by time
100  std::stable_sort(perm.begin(), perm.end(),
101  [&](std::size_t i, std::size_t j) { return isvect[i] < isvect[j]; });
102  // now apply permutation to time and gid output vectors
103  std::transform(perm.begin(), perm.end(), osvect.begin(),
104  [&](std::size_t i) { return isvect[i]; });
105  std::transform(perm.begin(), perm.end(), osvecg.begin(),
106  [&](std::size_t i) { return isvecg[i]; });
107 }

111 void sort_spikes(std::vector<double>& spikevec_time, std::vector<int>& spikevec_gid) {
112  double lmin_time = std::numeric_limits<double>::max();
113  double lmax_time = std::numeric_limits<double>::min();
114  if (!spikevec_time.empty()) {
115  lmin_time = *(std::min_element(spikevec_time.begin(), spikevec_time.end()));
116  lmax_time = *(std::max_element(spikevec_time.begin(), spikevec_time.end()));
117  }
118  double min_time = nrnmpi_dbl_allmin(lmin_time);
119  double max_time = nrnmpi_dbl_allmax(lmax_time);
120 
121  // allocate send and receive counts and displacements for MPI_Alltoallv
122  std::vector<int> snd_cnts(nrnmpi_numprocs);
123  std::vector<int> rcv_cnts(nrnmpi_numprocs);
124  std::vector<int> snd_dsps(nrnmpi_numprocs);
125  std::vector<int> rcv_dsps(nrnmpi_numprocs);
126 
127  double bin_t = (max_time - min_time) / nrnmpi_numprocs;
128  bin_t = bin_t ? bin_t : 1;
129  // first find number of spikes in each time window
130  for (const auto& st : spikevec_time) {
131  int idx = (int)(st - min_time) / bin_t;
132  snd_cnts[idx]++;
133  }
134  for (int i = 1; i < nrnmpi_numprocs; i++) {
135  snd_dsps[i] = snd_dsps[i - 1] + snd_cnts[i - 1];
136  }
137 
138  // now let each rank know how many spikes they will receive
139  // and get in turn all the buffer sizes to receive
140  nrnmpi_int_alltoall(&snd_cnts[0], &rcv_cnts[0], 1);
141  for (int i = 1; i < nrnmpi_numprocs; i++) {
142  rcv_dsps[i] = rcv_dsps[i - 1] + rcv_cnts[i - 1];
143  }
144  std::size_t new_sz = 0;
145  for (const auto& r : rcv_cnts) {
146  new_sz += r;
147  }
148  // prepare new sorted vectors
149  std::vector<double> svt_buf(new_sz, 0.0);
150  std::vector<int> svg_buf(new_sz, 0);
151 
152  // now exchange data
153  nrnmpi_dbl_alltoallv(spikevec_time.data(), &snd_cnts[0], &snd_dsps[0], svt_buf.data(),
154  &rcv_cnts[0], &rcv_dsps[0]);
155  nrnmpi_int_alltoallv(spikevec_gid.data(), &snd_cnts[0], &snd_dsps[0], svg_buf.data(),
156  &rcv_cnts[0], &rcv_dsps[0]);
157 
158  local_spikevec_sort(svt_buf, svg_buf, spikevec_time, spikevec_gid);
159 }

161 /** Write generated spikes to out.dat using mpi parallel i/o.
162  * \todo : MPI related code should be factored into nrnmpi.c
163  * Check spike record length which is set to 64 chars
164  */
165 void output_spikes_parallel(const char* outpath, const std::string& population_name) {
166  std::stringstream ss;
167  ss << outpath << "/out.dat";
168  std::string fname = ss.str();
169 
170  // remove if file already exist
171  if (nrnmpi_myid == 0) {
172  remove(fname.c_str());
173  }
174 #ifdef ENABLE_SONATA_REPORTS
175  sonata_write_spikes(population_name.data(), spikevec_time.data(), spikevec_time.size(), spikevec_gid.data(),
176  spikevec_gid.size(), outpath);
177 #endif // ENABLE_SONATA_REPORTS
178 
179  sort_spikes(spikevec_time, spikevec_gid);
180  nrnmpi_barrier();
181 
182  // each spike record in the file is time + gid (64 chars sufficient)
183  const int SPIKE_RECORD_LEN = 64;
184  unsigned num_spikes = spikevec_gid.size();
185  unsigned num_bytes = (sizeof(char) * num_spikes * SPIKE_RECORD_LEN);
186  char* spike_data = (char*)malloc(num_bytes);
187 
188  if (spike_data == nullptr) {
189  printf("Error while writing spikes due to memory allocation\n");
190  return;
191  }
192 
193  // empty if no spikes
194  strcpy(spike_data, "");
195 
196  // populate buffer with all spike entries
197  char spike_entry[SPIKE_RECORD_LEN];
198  unsigned spike_data_offset = 0;
199  for (unsigned i = 0; i < num_spikes; i++) {
200  int spike_entry_chars = snprintf(spike_entry, 64, "%.8g\t%d\n", spikevec_time[i], spikevec_gid[i]);
201  spike_data_offset = strcat_at_pos(spike_data, spike_data_offset, spike_entry, spike_entry_chars);
202  }
203 
204  // calculate offset into global file. note that we don't write
205  // all num_bytes but only "populated" buffer
206  unsigned long num_chars = strlen(spike_data);
207  unsigned long offset = 0;
208 
209  // global offset into file
210  MPI_Exscan(&num_chars, &offset, 1, MPI_UNSIGNED_LONG, MPI_SUM, MPI_COMM_WORLD);
211 
212  // write to file using parallel mpi i/o
213  MPI_File fh;
214  MPI_Status status;
215 
216  // ibm mpi (bg-q) expects char* instead of const char* (even though it's standard)
217  int op_status = MPI_File_open(MPI_COMM_WORLD, (char*)fname.c_str(),
218  MPI_MODE_CREATE | MPI_MODE_WRONLY, MPI_INFO_NULL, &fh);
219  if (op_status != MPI_SUCCESS && nrnmpi_myid == 0) {
220  std::cerr << "Error while opening spike output file " << fname << std::endl;
221  abort();
222  }
223 
224  op_status = MPI_File_write_at_all(fh, offset, spike_data, num_chars, MPI_BYTE, &status);
225  if (op_status != MPI_SUCCESS && nrnmpi_myid == 0) {
226  std::cerr << "Error while writing spike output " << std::endl;
227  abort();
228  }
229 
230  MPI_File_close(&fh);
231  free(spike_data);
232 }
