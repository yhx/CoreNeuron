76  * Create arrays for Jacobian, variable increments, function values, and
77  * permutation vector
78  */

85  error = EXCEED_ITERS;
86  done = 2;
87  }

95  for (int i = 0; i < n; i++)

97  * function values */
98  error = nrn_crout_thread(ns, n, jacobian, perm, _threadargs_);
99  if (error != SUCCESS) {
100  done = 2;
101  }
102  }
103 
104  if (!done) {
105  nrn_scopmath_solve_thread(n, jacobian, value, perm, delta_x, (int*)0, _threadargs_);
106 
107  /* Update solution vector and compute norms of delta_x and value */
108 
109  change = 0.0;
110  if (s) {
111  for (int i = 0; i < n; i++) {
112  if (fabs(s_(i)) > ZERO && (temp = fabs(delta_x[ix(i)] / (s_(i)))) > change)
113  change = temp;
114  s_(i) += delta_x[ix(i)];
115  }
116  } else {
117  for (int i = 0; i < n; i++) {
118  if (fabs(s_(i)) > ZERO && (temp = fabs(delta_x[ix(i)] / (s_(i)))) > change)
119  change = temp;
120  s_(i) += delta_x[ix(i)];
121  }
122  }
123  newtfun(pfunc); /* Evaluate function values with new solution */
124  max_dev = 0.0;
125  for (int i = 0; i < n; i++) {
126  value[ix(i)] = -value[ix(i)]; /* Required correction to function
127  * values */
128  if ((temp = fabs(value[ix(i)])) > max_dev)
129  max_dev = temp;
130  }
131 
132  /* Check for convergence or maximum iterations */
133 
134  if (change <= CONVERGE && max_dev <= ZERO) {
135  // break;
136  done = 1;
137  }
138  }
139  } /* end of while loop */
140 
141  return (error);
142 }

144 /*------------------------------------------------------------*/
145 /* */
146 /* BUILDJACOBIAN */
147 /* */
148 /* Creates the Jacobian matrix by computing partial deriv- */
149 /* atives by finite central differences. If the column */
150 /* variable is nonzero, an increment of 2% of the variable */
151 /* is used. STEP is the minimum increment allowed; it is */
152 /* currently set to 1.0E-6. */
153 /* */
154 /* Returns: no return variable */
155 /* */
156 /* Calling sequence: */
157 /* buildjacobian(n, index, x, pfunc, value, jacobian) */
158 /* */
159 /* Arguments: */
160 /* */
161 /* Input: n, integer, number of variables */
162 /* */
163 /* x, pointer to array of addresses of the solution */
164 /* vector elements */
165 /* */
166 /* p, array of parameter values */
167 /* */
168 /* pfunc, pointer to function which computes the */
169 /* deviation from zero of each equation */
170 /* in the model. */
171 /* */
172 /* value, pointer to array of addresses of function */
173 /* values */
174 /* */
175 /* Output: jacobian, double, computed jacobian matrix */
176 /* */
177 /* Functions called: user-supplied function with argument */
178 /* (p) to compute vector of function */
179 /* values for each equation. */
180 /* makevector(), freevector() */
181 /* */
182 /*------------------------------------------------------------*/

186 void nrn_buildjacobian_thread(NewtonSpace* ns,
187  int n,
188  int* index,
189  NEWTFUN pfunc,
190  double* value,
191  double** jacobian,
192  _threadargsproto_) {
193 #define x_(arg) _p[(arg)*_STRIDE]
194  double* high_value = ns->high_value;
195  double* low_value = ns->low_value;
196 
197  /* Compute partial derivatives by central finite differences */
198 
199  for (int j = 0; j < n; j++) {
200  double increment = max(fabs(0.02 * (x_(index[j]))), STEP);
201  x_(index[j]) += increment;
202  newtfun(pfunc);
203  for (int i = 0; i < n; i++)
204  high_value[ix(i)] = value[ix(i)];
205  x_(index[j]) -= 2.0 * increment;
206  newtfun(pfunc);
207  for (int i = 0; i < n; i++) {
208  low_value[ix(i)] = value[ix(i)];
209 
210  /* Insert partials into jth column of Jacobian matrix */
211 
212  jacobian[i][ix(j)] = (high_value[ix(i)] - low_value[ix(i)]) / (2.0 * increment);
213  }
214 
215  /* Restore original variable and function values. */
216 
217  x_(index[j]) += increment;
218  newtfun(pfunc);
219  }
220 }
