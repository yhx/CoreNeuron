155 /*
156  * The following code implements the basic operations on
157  * an event-set or priority-queue implemented using splay trees:
158  *
159 Hines changed to void spinit(SPTREE**) for use with TQueue.
160  * SPTREE *spinit( compare ) Make a new tree
161  * SPBLK *spenq( n, q ) Insert n in q after all equal keys.
162  * SPBLK *spdeq( np ) Return first key under *np, removing it.
163  * void splay( n, q ) n (already in q) becomes the root.
164  * int n = sphead( q ) n is the head item in q (not removed).
165  * spdelete( n, q ) n is removed from q.
166  *
167  * In the above, n points to an SPBLK type, while q points to an
168  * SPTREE.
169  *
170  * The implementation used here is based on the implementation
171  * which was used in the tests of splay trees reported in:
172  *
173  * An Empirical Comparison of Priority-Queue and Event-Set Implementations,
174  * by Douglas W. Jones, Comm. ACM 29, 4 (Apr. 1986) 300-311.
175  *
176  * The changes made include the addition of the enqprior
177  * operation and the addition of up-links to allow for the splay
178  * operation. The basic splay tree algorithms were originally
179  * presented in:
180  *
181  * Self Adjusting Binary Trees,
182  * by D. D. Sleator and R. E. Tarjan,
183  * Proc. ACM SIGACT Symposium on Theory
184  * of Computing (Boston, Apr 1983) 235-245.
185  *
186  * The enq and enqprior routines use variations on the
187  * top-down splay operation, while the splay routine is bottom-up.
188  * All are coded for speed.
189  *
190  * Written by:
191  * Douglas W. Jones
192  *
193  * Translated to C by:
194  * David Brower, daveb@rtech.uucp
195  *
196  * Thu Oct 6 12:11:33 PDT 1988 (daveb) Fixed spdeq, which was broken
197  * handling one-node trees. I botched the pascal translation of
198  * a VAR parameter.
199  */

211 /*----------------
212  *
213  * spenq() -- insert item in a tree.
214  *
215  * put n in q after all other nodes with the same key; when this is
216  * done, n will be the root of the splay tree representing q, all nodes
217  * in q with keys less than or equal to that of n will be in the
218  * left subtree, all with greater keys will be in the right subtree;
219  * the tree is split into these subtrees from the top down, with rotations
220  * performed along the way to shorten the left branch of the right subtree
221  * and the right branch of the left subtree
222  */
223 SPBLK* spenq(SPBLK* n, SPTREE* q) {
224  SPBLK* left; /* the rightmost node in the left tree */
225  SPBLK* right; /* the leftmost node in the right tree */
226  SPBLK* next; /* the root of the unsplit part */
227  SPBLK* temp;
228 
229  double key;
230 
231  n->uplink = nullptr;
232  next = q->root;
233  q->root = n;
234  if (next == nullptr) /* trivial enq */
235  {
236  n->leftlink = nullptr;
237  n->rightlink = nullptr;
238  } else /* difficult enq */
239  {
240  key = n->key;
241  left = n;
242  right = n;
243 
244  /* n's left and right children will hold the right and left
245  splayed trees resulting from splitting on n->key;
246  note that the children will be reversed! */
247 
248  q->enqcmps++;
249  if (STRCMP(next->key, key) > 0)
250  goto two;
251 
252  one: /* assert next->key <= key */
253 
254  do /* walk to the right in the left tree */
255  {
256  temp = next->rightlink;
257  if (temp == nullptr) {
258  left->rightlink = next;
259  next->uplink = left;
260  right->leftlink = nullptr;
261  goto done; /* job done, entire tree split */
262  }
263 
264  q->enqcmps++;
265  if (STRCMP(temp->key, key) > 0) {
266  left->rightlink = next;
267  next->uplink = left;
268  left = next;
269  next = temp;
270  goto two; /* change sides */
271  }
272 
273  next->rightlink = temp->leftlink;
274  if (temp->leftlink != nullptr)
275  temp->leftlink->uplink = next;
276  left->rightlink = temp;
277  temp->uplink = left;
278  temp->leftlink = next;
279  next->uplink = temp;
280  left = temp;
281  next = temp->rightlink;
282  if (next == nullptr) {
283  right->leftlink = nullptr;
284  goto done; /* job done, entire tree split */
285  }
286 
287  q->enqcmps++;
288 
289  } while (STRCMP(next->key, key) <= 0); /* change sides */
290 
291  two: /* assert next->key > key */
292 
293  do /* walk to the left in the right tree */
294  {
295  temp = next->leftlink;
296  if (temp == nullptr) {
297  right->leftlink = next;
298  next->uplink = right;
299  left->rightlink = nullptr;
300  goto done; /* job done, entire tree split */
301  }
302 
303  q->enqcmps++;
304  if (STRCMP(temp->key, key) <= 0) {
305  right->leftlink = next;
306  next->uplink = right;
307  right = next;
308  next = temp;
309  goto one; /* change sides */
310  }
311  next->leftlink = temp->rightlink;
312  if (temp->rightlink != nullptr)
313  temp->rightlink->uplink = next;
314  right->leftlink = temp;
315  temp->uplink = right;
316  temp->rightlink = next;
317  next->uplink = temp;
318  right = temp;
319  next = temp->leftlink;
320  if (next == nullptr) {
321  left->rightlink = nullptr;
322  goto done; /* job done, entire tree split */
323  }
324 
325  q->enqcmps++;
326 
327  } while (STRCMP(next->key, key) > 0); /* change sides */
328 
329  goto one;
330 
331  done: /* split is done, branches of n need reversal */
332 
333  temp = n->leftlink;
334  n->leftlink = n->rightlink;
335  n->rightlink = temp;
336  }
337 
338  return (n);
339 
340 } /* spenq */

342 /*----------------
343  *
344  * spdeq() -- return and remove head node from a subtree.
345  *
346  * remove and return the head node from the node set; this deletes
347  * (and returns) the leftmost node from q, replacing it with its right
348  * subtree (if there is one); on the way to the leftmost node, rotations
349  * are performed to shorten the left branch of the tree
350  */
351 SPBLK* spdeq(SPBLK** np) /* pointer to a node pointer */
352 
353 {
354  SPBLK* deq; /* one to return */
355  SPBLK* next; /* the next thing to deal with */
356  SPBLK* left; /* the left child of next */
357  SPBLK* farleft; /* the left child of left */
358  SPBLK* farfarleft; /* the left child of farleft */
359 
360  if (np == nullptr || *np == nullptr) {
361  deq = nullptr;
362  } else {
363  next = *np;
364  left = next->leftlink;
365  if (left == nullptr) {
366  deq = next;
367  *np = next->rightlink;
368 
369  if (*np != nullptr)
370  (*np)->uplink = nullptr;
371 
372  } else
373  for (;;) /* left is not null */
374  {
375  /* next is not it, left is not nullptr, might be it */
376  farleft = left->leftlink;
377  if (farleft == nullptr) {
378  deq = left;
379  next->leftlink = left->rightlink;
380  if (left->rightlink != nullptr)
381  left->rightlink->uplink = next;
382  break;
383  }
384 
385  /* next, left are not it, farleft is not nullptr, might be it */
386  farfarleft = farleft->leftlink;
387  if (farfarleft == nullptr) {
388  deq = farleft;
389  left->leftlink = farleft->rightlink;
390  if (farleft->rightlink != nullptr)
391  farleft->rightlink->uplink = left;
392  break;
393  }
394 
395  /* next, left, farleft are not it, rotate */
396  next->leftlink = farleft;
397  farleft->uplink = next;
398  left->leftlink = farleft->rightlink;
399  if (farleft->rightlink != nullptr)
400  farleft->rightlink->uplink = left;
401  farleft->rightlink = left;
402  left->uplink = farleft;
403  next = farleft;
404  left = farfarleft;
405  }
406  }
407 
408  return (deq);
409 
410 } /* spdeq */

412 /*----------------
413  *
414  * splay() -- reorganize the tree.
415  *
416  * the tree is reorganized so that n is the root of the
417  * splay tree representing q; results are unpredictable if n is not
418  * in q to start with; q is split from n up to the old root, with all
419  * nodes to the left of n ending up in the left subtree, and all nodes
420  * to the right of n ending up in the right subtree; the left branch of
421  * the right subtree and the right branch of the left subtree are
422  * shortened in the process
423  *
424  * this code assumes that n is not nullptr and is in q; it can sometimes
425  * detect n not in q and complain
426  */
427 
428 void splay(SPBLK* n, SPTREE* q) {
429  SPBLK* up; /* points to the node being dealt with */
430  SPBLK* prev; /* a descendent of up, already dealt with */
431  SPBLK* upup; /* the parent of up */
432  SPBLK* upupup; /* the grandparent of up */
433  SPBLK* left; /* the top of left subtree being built */
434  SPBLK* right; /* the top of right subtree being built */
435 
436  left = n->leftlink;
437  right = n->rightlink;
438  prev = n;
439  up = prev->uplink;
440 
441  while (up != nullptr) {
442  /* walk up the tree towards the root, splaying all to the left of
443  n into the left subtree, all to right into the right subtree */
444 
445  upup = up->uplink;
446  if (up->leftlink == prev) /* up is to the right of n */
447  {
448  if (upup != nullptr && upup->leftlink == up) /* rotate */
449  {
450  upupup = upup->uplink;
451  upup->leftlink = up->rightlink;
452  if (upup->leftlink != nullptr)
453  upup->leftlink->uplink = upup;
454  up->rightlink = upup;
455  upup->uplink = up;
456  if (upupup == nullptr)
457  q->root = up;
458  else if (upupup->leftlink == upup)
459  upupup->leftlink = up;
460  else
461  upupup->rightlink = up;
462  up->uplink = upupup;
463  upup = upupup;
464  }
465  up->leftlink = right;
466  if (right != nullptr)
467  right->uplink = up;
468  right = up;
469 
470  } else /* up is to the left of n */
471  {
472  if (upup != nullptr && upup->rightlink == up) /* rotate */
473  {
474  upupup = upup->uplink;
475  upup->rightlink = up->leftlink;
476  if (upup->rightlink != nullptr)
477  upup->rightlink->uplink = upup;
478  up->leftlink = upup;
479  upup->uplink = up;
480  if (upupup == nullptr)
481  q->root = up;
482  else if (upupup->rightlink == upup)
483  upupup->rightlink = up;
484  else
485  upupup->leftlink = up;
486  up->uplink = upupup;
487  upup = upupup;
488  }
489  up->rightlink = left;
490  if (left != nullptr)
491  left->uplink = up;
492  left = up;
493  }
494  prev = up;
495  up = upup;
496  }
497 
498 #ifdef DEBUG
499  if (q->root != prev) {
500  /* fprintf(stderr, " *** bug in splay: n not in q *** " ); */
501  abort();
502  }
503 #endif
504 
505  n->leftlink = left;
506  n->rightlink = right;
507  if (left != nullptr)
508  left->uplink = n;
509  if (right != nullptr)
510  right->uplink = n;
511  q->root = n;
512  n->uplink = nullptr;
513 
514 } /* splay */

516 /*----------------
517  *
518  * sphead() -- return the "lowest" element in the tree.
519  *
520  * returns a reference to the head event in the event-set q,
521  * represented as a splay tree; q->root ends up pointing to the head
522  * event, and the old left branch of q is shortened, as if q had
523  * been splayed about the head element; this is done by dequeueing
524  * the head and then making the resulting queue the right son of
525  * the head returned by spdeq; an alternative is provided which
526  * avoids splaying but just searches for and returns a pointer to
527  * the bottom of the left branch
528  */
529 SPBLK* sphead(SPTREE* q) {
530  SPBLK* x;
531 
532  /* splay version, good amortized bound */
533  x = spdeq(&q->root);
534  if (x != nullptr) {
535  x->rightlink = q->root;
536  x->leftlink = nullptr;
537  x->uplink = nullptr;
538  if (q->root != nullptr)
539  q->root->uplink = x;
540  }
541  q->root = x;
542 
543  /* alternative version, bad amortized bound,
544  but faster on the average */
545 
546  return (x);
547 
548 } /* sphead */

550 /*----------------
551  *
552  * spdelete() -- Delete node from a tree.
553  *

555  * around its new root, which is the successor of n

558 void spdelete(SPBLK* n, SPTREE* q) {

560

574  x->leftlink->uplink = x;
575  if (x->rightlink != nullptr)
