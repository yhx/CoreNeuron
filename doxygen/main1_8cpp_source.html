100 char* prepare_args(int& argc, char**& argv, int use_mpi, const char* arg) {
101  // first construct all arguments as string
102  std::string args(arg);
103  args.insert(0, " coreneuron ");
104  args.append(" --skip-mpi-finalize ");
105  if (use_mpi) {
106  args.append(" --mpi ");
107  }
108 
109  // we can't modify string with strtok, make copy
110  char* first = strdup(args.c_str());
111  const char* sep = " ";
112 
113  // first count the no of argument
114  char* token = strtok(first, sep);
115  argc = 0;
116  while (token) {
117  token = strtok(nullptr, sep);
118  argc++;
119  }
120  free(first);
121 
122  // now build char*argv
123  argv = new char*[argc];
124  first = strdup(args.c_str());
125  token = strtok(first, sep);
126  for (int i = 0; token; i++) {
127  argv[i] = token;
128  token = strtok(nullptr, sep);
129  }
130 
131  // return actual data to be freed
132  return first;
133 }

353 void get_nrn_trajectory_requests(int bsize) {
354  if (nrn2core_get_trajectory_requests_) {
355  for (int tid = 0; tid < nrn_nthread; ++tid) {
356  NrnThread& nt = nrn_threads[tid];
357  int n_pr;
358  int n_trajec;
359  int* types;
360  int* indices;
361  void** vpr;
362  double** varrays;
363  double** pvars;
364 
365  // bsize is passed by reference, the return value will determine if
366  // per step return or entire trajectory return.
367  (*nrn2core_get_trajectory_requests_)(tid, bsize, n_pr, vpr, n_trajec, types, indices,
368  pvars, varrays);
369  delete_trajectory_requests(nt);
370  if (n_trajec) {
371  TrajectoryRequests* tr = new TrajectoryRequests;
372  nt.trajec_requests = tr;
373  tr->bsize = bsize;
374  tr->n_pr = n_pr;
375  tr->n_trajec = n_trajec;
376  tr->vsize = 0;
377  tr->vpr = vpr;
378  tr->gather = new double*[n_trajec];
379  tr->varrays = varrays;
380  tr->scatter = pvars;
381  for (int i = 0; i < n_trajec; ++i) {
382  tr->gather[i] = stdindex2ptr(types[i], indices[i], nt);
383  }
384  delete[] types;
385  delete[] indices;
386  }
387  }
388  }
