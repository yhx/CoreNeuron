127 size_t read_raster_file(const char* fname, double** tvec, int** gidvec, double tstop) {
128  FILE* f = fopen(fname, "r");
129  nrn_assert(f);
130 
131  // skip first line containing "scatter" string
132  char dummy[100];
133  nrn_assert(fgets(dummy, 100, f));
134 
135  std::vector<spike_type> spikes;
136  spikes.reserve(10000);
137 
138  double stime;
139  int gid;
140 
141  while (fscanf(f, "%lf %d\n", &stime, &gid) == 2) {
142  if ( stime >= t && stime <= tstop) {
143  spikes.push_back(std::make_pair(stime, gid));
144  }
145  }
146 
147  fclose(f);
148 
149  // pattern.mod expects sorted spike raster (this is to avoid
150  // injecting all events at the begining of the simulation).
151  // sort spikes according to time
152  std::sort(spikes.begin(), spikes.end(), spike_comparator);
153 
154  // fill gid and time vectors
155  *tvec = (double*)emalloc(spikes.size() * sizeof(double));
156  *gidvec = (int*)emalloc(spikes.size() * sizeof(int));
157 
158  for (size_t i = 0; i < spikes.size(); i++) {
159  (*tvec)[i] = spikes[i].first;
160  (*gidvec)[i] = spikes[i].second;
161  }
162 
163  return spikes.size();
164 }

167 static NrnThreadMembList* alloc_nrn_thread_memb(int type) {
168  NrnThreadMembList* tml = (NrnThreadMembList*)emalloc(sizeof(NrnThreadMembList));
169  tml->dependencies = nullptr;
170  tml->ndependencies = 0;
171  tml->index = type;
172  tml->next = nullptr;
173 
174  // fill in tml->ml info. The data is not in the cache efficient
175  // NrnThread arrays but there should not be many of these instances.
176  int psize = corenrn.get_prop_param_size()[type];
177  int dsize = corenrn.get_prop_dparam_size()[type];
178 
179  tml->ml = (Memb_list*)emalloc(sizeof(Memb_list));
180  tml->ml->nodecount = 1;
181  tml->ml->_nodecount_padded = tml->ml->nodecount;
182  tml->ml->nodeindices = nullptr;
183  tml->ml->data = (double*)ecalloc(tml->ml->nodecount * psize, sizeof(double));
184  tml->ml->pdata = (Datum*)ecalloc(tml->ml->nodecount * dsize, sizeof(Datum));
185  tml->ml->_thread = nullptr;
186  tml->ml->_net_receive_buffer = nullptr;
187  tml->ml->_net_send_buffer = nullptr;
188  tml->ml->_permute = nullptr;
189 
190  return tml;
191 }

200 Point_process* nrn_artcell_instantiate(const char* mechname) {
201  int type = nrn_get_mechtype(mechname);
202  NrnThread* nt = nrn_threads + 0;
203 
204  // printf("nrn_artcell_instantiate %s type=%d\n", mechname, type);
205 
206  // create and append to nt.tml
207  auto tml = alloc_nrn_thread_memb(type);
208 
209  assert(nt->_ml_list[type] == nullptr); // FIXME
210  nt->_ml_list[type] = tml->ml;
211 
212  if (!nt->tml) {
213  nt->tml = tml;
214  } else {
215  for (NrnThreadMembList* i = nt->tml; i; i = i->next) {
216  if (!i->next) {
217  i->next = tml;
218  break;
219  }
220  }
221  }

223  // Here we have a problem with no easy general solution. ml->pdata are
224  // integer indexes into the nt->_data nt->_idata and nt->_vdata array
225  // depending on context,
226  // but nrn_setup.cpp allocated these to exactly have the size needed by
227  // the file defined model (at least for _vdata) and so there are no slots
228  // for pdata to index into for this new instance.
229  // So nrn_setup.cpp:phase2 needs to
230  // be notified that some extra space will be required. For now, defer
231  // the general situation of several instances for several types and
232  // demand that this method is never called more than once. We introduce
233  // a int nrn_extra_thread0_vdata (only that is needed by PatternStim)
234  // which will be used by
235  // nrn_setup.cpp:phase2 to allocate the appropriately larger
236  // _vdata arrays for thread 0 (without changing _nvdata so
237  // that we can fill in the indices here)
238  static int cnt = 0;
239  if (++cnt > 1) {
240  printf("nrn_artcell_instantiate cannot be called more than once\n");
241  assert(0);
242  }
243  // note that PatternStim internal usage for the 4 ppvar values is:
244  // #define _nd_area _nt->_data[_ppvar[0]] (not used since ARTIFICIAL_CELL)
245  // #define _p_ptr _nt->_vdata[_ppvar[2]] (the BBCORE_POINTER)
246  // #define _tqitem &(_nt->_vdata[_ppvar[3]]) (for net_send)
247  // and general external usage is:
248  // _nt->_vdata[_ppvar[1]] = Point_process*
249  //
250 
251  Point_process* pnt = new Point_process;
252  pnt->_type = type;
253  pnt->_tid = nt->id;
254  pnt->_i_instance = 0;
255  // as though all dparam index into _vdata
256  int dsize = corenrn.get_prop_dparam_size()[type];
257  assert(dsize <= nrn_extra_thread0_vdata);
258  for (int i = 0; i < dsize; ++i) {
259  tml->ml->pdata[i] = nt->_nvdata + i;
260  }
261  nt->_vdata[nt->_nvdata + 1] = (void*)pnt;
262 
263  return pnt;
264 }
