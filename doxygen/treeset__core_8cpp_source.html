43 static void nrn_rhs(NrnThread* _nt) {
44  int i1 = 0;
45  int i2 = i1 + _nt->ncell;
46  int i3 = _nt->end;
47 
48  double* vec_rhs = &(VEC_RHS(0));
49  double* vec_d = &(VEC_D(0));
50  double* vec_a = &(VEC_A(0));
51  double* vec_b = &(VEC_B(0));
52  double* vec_v = &(VEC_V(0));
53  int* parent_index = _nt->_v_parent_index;
54 
55 // clang-format off
56  #pragma acc parallel loop present( \
57  vec_rhs[0:i3], vec_d[0:i3]) \
58  if (_nt->compute_gpu) async(_nt->stream_id)
59  // clang-format on
60  for (int i = i1; i < i3; ++i) {
61  vec_rhs[i] = 0.;
62  vec_d[i] = 0.;
63  }
64 
65  if (_nt->nrn_fast_imem) {
66  for (int i = i1; i < i3; ++i) {
67  _nt->nrn_fast_imem->nrn_sav_rhs[i] = 0.;
68  _nt->nrn_fast_imem->nrn_sav_d[i] = 0.;
69  }
70  }
71 
72  nrn_ba(_nt, BEFORE_BREAKPOINT);
73  /* note that CAP has no current */
74  for (auto tml = _nt->tml; tml; tml = tml->next)
75  if (corenrn.get_memb_func(tml->index).current) {
76  mod_f_t s = corenrn.get_memb_func(tml->index).current;
77  std::string ss("cur-");
78  ss += nrn_get_mechname(tml->index);
79  Instrumentor::phase p(ss.c_str());
80  (*s)(_nt, tml->ml, tml->index);
81 #ifdef DEBUG
82  if (errno) {
83  hoc_warning("errno set during calculation of currents", nullptr);
84  }
85 #endif
86  }
87 
88  if (_nt->nrn_fast_imem) {
89  /* _nrn_save_rhs has only the contribution of electrode current
90  so here we transform so it only has membrane current contribution
91  */
92  double* p = _nt->nrn_fast_imem->nrn_sav_rhs;
93  for (int i = i1; i < i3; ++i) {
94  p[i] -= vec_rhs[i];
95  }
96  }
97 
98 /* now the internal axial currents.
99 The extracellular mechanism contribution is already done.
100  rhs += ai_j*(vi_j - vi)
101 */
102 // clang-format off
103  #pragma acc parallel loop present( \
104  vec_rhs[0:i3], vec_d[0:i3], \
105  vec_a[0:i3], vec_b[0:i3], \
106  vec_v[0:i3], parent_index[0:i3]) \
107  if (_nt->compute_gpu) async(_nt->stream_id)
108  for (int i = i2; i < i3; ++i) {
109  double dv = vec_v[parent_index[i]] - vec_v[i];
110  /* our connection coefficients are negative so */
111  #pragma acc atomic update
112  vec_rhs[i] -= vec_b[i] * dv;
113  #pragma acc atomic update
114  vec_rhs[parent_index[i]] += vec_a[i] * dv;
115  }
116  // clang-format on
117 }
118 
119 /* calculate left hand side of
120 cm*dvm/dt = -i(vm) + is(vi) + ai_j*(vi_j - vi)
121 cx*dvx/dt - cm*dvm/dt = -gx*(vx - ex) + i(vm) + ax_j*(vx_j - vx)
122 with a matrix so that the solution is of the form [dvm+dvx,dvx] on the right
123 hand side after solving.
124 This is a common operation for fixed step, cvode, and daspk methods
125 */

127 static void nrn_lhs(NrnThread* _nt) {
128  int i1 = 0;
129  int i2 = i1 + _nt->ncell;
130  int i3 = _nt->end;
131 
132  /* note that CAP has no jacob */
133  for (auto tml = _nt->tml; tml; tml = tml->next)
134  if (corenrn.get_memb_func(tml->index).jacob) {
135  mod_f_t s = corenrn.get_memb_func(tml->index).jacob;
136  std::string ss("cur-");
137  ss += nrn_get_mechname(tml->index);
138  Instrumentor::phase p(ss.c_str());
139  (*s)(_nt, tml->ml, tml->index);
140 #ifdef DEBUG
141  if (errno) {
142  hoc_warning("errno set during calculation of jacobian", (char*)0);
143  }
144 #endif
145  }
146  /* now the cap current can be computed because any change to cm by another model
147  has taken effect
148  */
149  /* note, the first is CAP if there are any nodes*/
150  if (_nt->end && _nt->tml) {
151  assert(_nt->tml->index == CAP);
152  nrn_jacob_capacitance(_nt, _nt->tml->ml, _nt->tml->index);
153  }
154 
155  double* vec_d = &(VEC_D(0));
156  double* vec_a = &(VEC_A(0));
157  double* vec_b = &(VEC_B(0));
158  int* parent_index = _nt->_v_parent_index;
159 
160  if (_nt->nrn_fast_imem) {
161  /* _nrn_save_d has only the contribution of electrode current
162  so here we transform so it only has membrane current contribution
163  */
164  double* p = _nt->nrn_fast_imem->nrn_sav_d;
165  for (int i = i1; i < i3; ++i) {
166  p[i] += vec_d[i];
167  }
168  }
169 
170 /* now add the axial currents */
171 // clang-format off
172  #pragma acc parallel loop present( \
173  vec_d[0:i3], vec_a[0:i3], \
174  vec_b[0:i3], parent_index[0:i3]) \
175  if (_nt->compute_gpu) async(_nt->stream_id)
176  for (int i = i2; i < i3; ++i) {
177  #pragma acc atomic update
178  vec_d[i] -= vec_b[i];
179  #pragma acc atomic update
180  vec_d[parent_index[i]] -= vec_a[i];
181  }
182  // clang-format on
183 }
