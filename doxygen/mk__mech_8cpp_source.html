99 static void mk_mech(std::istream& s) {
100  char version[256];
101  s >> version;
102  check_bbcore_write_version(version);
103 
104  // printf("reading %s\n", fname);
105  int n = 0;
106  nrn_assert(s >> n);
107 
108  /// Allocate space for mechanism related data structures
109  alloc_mech(n);
110 
111  /// Read all the mechanisms and their meta data
112  for (int i = 2; i < n; ++i) {
113  char mname[100];
114  int type = 0, pnttype = 0, is_art = 0, is_ion = 0, dsize = 0, pdsize = 0;
115  nrn_assert(s >> mname >> type >> pnttype >> is_art >> is_ion >> dsize >> pdsize);
116  nrn_assert(i == type);
117 #ifdef DEBUG
118  printf("%s %d %d %d %d %d %d\n", mname, type, pnttype, is_art, is_ion, dsize, pdsize);
119 #endif
120  std::string str(mname);
121  corenrn.get_memb_func(type).sym = (Symbol*)strdup(mname);
122  mech2type[str] = type;
123  corenrn.get_pnt_map()[type] = (char)pnttype;
124  corenrn.get_prop_param_size()[type] = dsize;
125  corenrn.get_prop_dparam_size()[type] = pdsize;
126  corenrn.get_is_artificial()[type] = is_art;
127  if (is_ion) {

130  // strip the _ion
131  char iname[100];
132  strcpy(iname, mname);

134  // printf("%s %s\n", mname, iname);
