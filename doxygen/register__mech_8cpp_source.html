236 void hoc_register_dparam_semantics(int type, int ix, const char* name) {
237  /* needed for SoA to possibly reorder name_ion and some "pointer" pointers. */
238  /* only interested in area, iontype, cvode_ieq,
239  netsend, pointer, pntproc, bbcorepointer, watch, diam, fornetcon
240  xx_ion and #xx_ion which will get
241  a semantics value of -1, -2, -3,
242  -4, -5, -6, -7, -8, -9, -10,
243  type, and type+1000 respectively
244  */
245  auto& memb_func = corenrn.get_memb_funcs();
246  if (strcmp(name, "area") == 0) {
247  memb_func[type].dparam_semantics[ix] = -1;
248  } else if (strcmp(name, "iontype") == 0) {
249  memb_func[type].dparam_semantics[ix] = -2;
250  } else if (strcmp(name, "cvodeieq") == 0) {
251  memb_func[type].dparam_semantics[ix] = -3;
252  } else if (strcmp(name, "netsend") == 0) {
253  memb_func[type].dparam_semantics[ix] = -4;
254  } else if (strcmp(name, "pointer") == 0) {
255  memb_func[type].dparam_semantics[ix] = -5;
256  } else if (strcmp(name, "pntproc") == 0) {
257  memb_func[type].dparam_semantics[ix] = -6;
258  } else if (strcmp(name, "bbcorepointer") == 0) {
259  memb_func[type].dparam_semantics[ix] = -7;
260  } else if (strcmp(name, "watch") == 0) {
261  memb_func[type].dparam_semantics[ix] = -8;
262  } else if (strcmp(name, "diam") == 0) {
263  memb_func[type].dparam_semantics[ix] = -9;
264  } else if (strcmp(name, "fornetcon") == 0) {
265  memb_func[type].dparam_semantics[ix] = -10;
266  } else {
267  int i = name[0] == '#' ? 1 : 0;
268  int etype = nrn_get_mechtype(name + i);
269  memb_func[type].dparam_semantics[ix] = etype + i * 1000;
270  /* note that if style is needed (i==1), then we are writing a concentration */
271  if (i) {
272  ion_write_depend(type, etype);
273  }
274  }
275 #if DEBUG
276  printf("dparam semantics %s ix=%d %s %d\n", memb_func[type].sym,
277  ix, name, memb_func[type].dparam_semantics[ix]);
278 #endif
279 }

299 static int depend_append(int idep, int* dependencies, int deptype, int type) {
300  /* append only if not already in dependencies and != type*/
301  bool add = true;
302  if (deptype == type) {
303  return idep;
304  }
305  for (int i = 0; i < idep; ++i) {
306  if (deptype == dependencies[i]) {
307  add = false;
308  break;
309  }
310  }
311  if (add) {
312  dependencies[idep++] = deptype;
313  }
314  return idep;
315 }

317 /* return list of types that this type depends on (10 should be more than enough) */
318 /* dependencies must be an array that is large enough to hold that array */
319 /* number of dependencies is returned */
320 int nrn_mech_depend(int type, int* dependencies) {
321  int dpsize = corenrn.get_prop_dparam_size()[type];
322  int* ds = corenrn.get_memb_func(type).dparam_semantics;
323  int idep = 0;
324  if (ds)
325  for (int i = 0; i < dpsize; ++i) {
326  if (ds[i] > 0 && ds[i] < 1000) {
327  int deptype = ds[i];
328  int idepnew = depend_append(idep, dependencies, deptype, type);
329  if ((idepnew > idep)
330  && !corenrn.get_ion_write_dependency().empty()
331  && !corenrn.get_ion_write_dependency()[deptype].empty()) {
332  auto& iwd = corenrn.get_ion_write_dependency()[deptype];
333  int size = iwd[0];
334  for (int j = 1; j < size; ++j) {
335  idepnew = depend_append(idepnew, dependencies, iwd[j], type);
336  }
337  }
338  idep = idepnew;
339  }
340  }
341  return idep;
342 }
